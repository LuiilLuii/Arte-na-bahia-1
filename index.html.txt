        let currentFilter = 'all';
        let searchQuery = '';

        let favorites = JSON.parse(localStorage.getItem('favorites') || '[]');

        function saveFavorites() {
            localStorage.setItem('favorites', JSON.stringify(favorites));
        }

        function toggleFavorite(id) {
            const index = favorites.indexOf(id);
            if (index > -1) {
                favorites.splice(index, 1);
                showToast('Removido dos favoritos', 'info');
            } else {
                favorites.push(id);
                showToast('Adicionado aos favoritos! ❤️', 'success');
            }
            saveFavorites();
            renderOpportunities();
        }

        function filterFavorites() {
            currentFilter = 'favorites';
            document.querySelectorAll('.tab-btn').forEach(btn => btn.classList.remove('active'));
            const favBtn = Array.from(document.querySelectorAll('.tab-btn')).find(btn => btn.innerText.includes('Favoritos'));
            if (favBtn) favBtn.classList.add('active');
            renderOpportunities();
        }

        function calculateDaysLeft(deadline) {
            if (deadline === 'contínuo') return -999;
            const today = new Date('2026-05-17');
            const deadlineDate = new Date(deadline);
            return Math.ceil((deadlineDate.getTime() - today.getTime()) / (1000 * 3600 * 24));
        }

        function createOpportunityCard(opp) {
            const daysLeft = calculateDaysLeft(opp.deadline);
            const isFav = favorites.includes(opp.id);

            let typeConfig = {
                'edital': { label: 'EDITAL', color: 'bg-[#118ab2] text-white' },
                'residencia': { label: 'RESIDÊNCIA', color: 'bg-[#9d4edd] text-white' },

                'recurso': { label: 'RECURSO', color: 'bg-[#bc6c25] text-white' }
            };

            const cfg = typeConfig[opp.type];
            let urgencyVisual = '';
            if (daysLeft === -999) {
                urgencyVisual = '<span class="bg-[#bc6c25] text-white px-3 py-1 rounded-full text-xs font-bold border-2 border-[#1a1a1a]">Sempre disponível</span>';
            } else if (daysLeft < 0) {
                urgencyVisual = '<span class="bg-[#1a1a1a] text-white px-3 py-1 rounded-full text-xs font-bold border-2 border-[#1a1a1a]">Encerrado</span>';
            } else {

            }

            return `
                <div class="naif-border flex flex-col h-full overflow-hidden">
                    <div class="${cfg.color} p-4 border-b-3 border-[#1a1a1a] flex justify-between items-center">
                        <span class="font-display text-sm tracking-wider">${cfg.label}</span>

        function renderOpportunities() {
            const grid = document.getElementById('opportunities-grid');
            let filtered = mockOpportunities.filter(opp => {
                if (currentFilter === 'favorites') {
                    return favorites.includes(opp.id);
                }
                if (currentFilter !== 'all' && opp.type !== currentFilter) return false;
                if (searchQuery) {
                    const q = searchQuery.toLowerCase();
                    if (!opp.title.toLowerCase().includes(q) && !opp.description.toLowerCase().includes(q)) return false;
                }
                if (regionQuery !== 'all' && opp.region !== regionQuery && opp.region !== 'todas') return false;
                if (categoryQuery !== 'all' && opp.category !== categoryQuery) return false;
                return true;
            });
            
            if(filtered.length === 0) {
                grid.innerHTML = `
                <div class="col-span-full py-16 text-center">
                    <div class="inline-block bg-white naif-border p-8">
                        <i data-lucide="frown" class="w-16 h-16 mx-auto mb-4 text-[#ef476f]"></i>
                        <h2 class="font-display text-2xl text-[#1a1a1a] mb-2">Vixe! Nada encontrado.</h2>
                        <p class="text-lg font-bold text-gray-600">Tente mudar os filtros ou adicionar favoritos.</p>
                    </div>
                </div>`;
            } else {
                grid.innerHTML = filtered.map(createOpportunityCard).join('');
            }
            lucide.createIcons();
        }

        function showToast(message, type = 'success') {
            const container = document.getElementById('toast-container');
            const toast = document.createElement('div');
            const color = type === 'success' ? 'bg-[#008c4a]' : type === 'info' ? 'bg-[#118ab2]' : 'bg-[#1a1a1a]';
            toast.className = `${color} text-white px-6 py-4 naif-border flex items-center transform transition-all duration-300 translate-y-10 opacity-0`;
            toast.innerHTML = `<i data-lucide="bell" class="w-6 h-6 mr-3 text-[#ffb703]"></i><span class="font-display text-lg tracking-wide">${message}</span>`;
            container.appendChild(toast);
            lucide.createIcons();
            setTimeout(() => toast.classList.remove('translate-y-10', 'opacity-0'), 10);
            setTimeout(() => { toast.classList.add('translate-y-10', 'opacity-0'); setTimeout(() => toast.remove(), 300); }, 4000);
        }
